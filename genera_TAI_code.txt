Apoc_sc <- readRDS(file = "Apoc_embryo_sc_final.rds")
library(Seurat)
install.packages("myTAI", dependencies = TRUE)
library(myTAI)
library(tibble)
library(ggplot2)

phylostrata_Ap <- read.table("165419_gene_ages.tsv",sep = "\t")
#change the column names
colnames(phylostrata_Ap) <- c("GeneID","name_PS","phylostratum","score")
#calculate the average expression across each of the clusters
Ap_average <- log1p(AverageExpression(Apoc_sc)$RNA)
Ap_average <- rownames_to_column(as.data.frame(Ap_average), "GeneID")
#omits any NAs in the data
phylostrata_Ap <- na.omit(phylostrata_Ap)
#subsets only genes that are in the phylostrata data
Ap_average <- subset(Ap_average, subset = Ap_average$GeneID %in% phylostrata_Ap$GeneID)
Phylo_expr_Ap <- match_map(Ap_average, phylostrata_Ap[,c("phylostratum","GeneID")])
TAI <- as.data.frame(TAI(Phylo_expr_Ap))
TAI <- rownames_to_column(TAI, var= "cluster")
colnames(TAI) <- c("cluster","TAI")
custom_colors <- c("nem.1" = "#EC8239", "nem.2" = "#64B200", "spirocytes" = "#B385FF", "mat.cnidos" = "#FF6B94", "cnido.progen" = "#00BFC4", "neural" = "#7B96FF", "secretory.3" = "#D874FD", "neurogland.progen" = "#00B81B", "mucous.gland" = "#AEA200", "gland.mouth" = "#00C085", "secretory.1" = "#00A6FF", "gland.1" = "#FD61D3", "secretory.2" = "#F8766D", "blastula" = "#EF67EB", "progen.1" = "#00BD5C", "progen.2" = "#FF63B6", "sec.progen.1" = "#DB8E00", "sec.progen.2" = "#00B2F3", "endoderm.progen" = "#00C1A7", "gastrodermis" = "#00BADE")
Apoc_TAI <- ggplot(TAI) + geom_point(aes(x=reorder(cluster,TAI), y=TAI, colour= factor(cluster)), size = 3) + RotatedAxis() +  theme(axis.text.x = element_text(angle = 90, vjust = 0.5, hjust = 1), legend.position = "none") +  scale_color_manual(values = custom_colors)
Apoc_TAI
