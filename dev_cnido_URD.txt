library(dplyr)library(plyr)library(Seurat)library(data.table)library(magrittr)require(scales)library(URD)library(cowplot)setwd("~/Documents/Astrangia/scRNAseq/Apoc_scRNAseq_cnido_subset_2-22-25")cnido_subset <- readRDS(file = "cnido_subset_2-22-25.rds")DimPlot(cnido_subset, reduction = "umap", label = "TRUE")

#proportion of each time point making up clusterscounts_table <- table(Idents(cnido_subset), cnido_subset$sample)print(counts_table)cluster_proportions <- prop.table(counts_table, margin = 1)print(cluster_proportions)#cluster 9 are early progenitors - cells from this cluster come almost exclusively from 24hpf
#subset out the early progenitor cellsprogen <- subset(cnido_subset, idents = c("9")) #subet out terminal cells - not going to use mature cnido cluster 1, but rather#the dev cnido clusters that show different identity based on type#using four clusters that represent basitrichs, mastigophores, robust spiros and graciles spirosterm_cells <- subset(cnido_subset, idents = c("3", "4", "5", "8"))#subset out intermediate cell clustersint_cells <- subset(cnido_subset, idents = c("6", "7", "10", "0", "2", "11"))#extract cell population data as matriciesterm_cells.matrix <- GetAssayData(object = term_cells, assay = "RNA", layer = "data")int_cells.matrix <- GetAssayData(object = int_cells, assay = "RNA", layer = "data")progen.matrix <- GetAssayData(object = progen, assay = "RNA", layer = "data")#extract cell population specific metadataterm_cells.metadata <- term_cells@meta.data#do following line 2 timesterm_cells.metadata <- term_cells.metadata[,-5]#do the following line 2 timesterm_cells.metadata <- term_cells.metadata[,-4]term_cells.metadata$orig.ident <- "terminal"names(term_cells.metadata)[names(term_cells.metadata) == "RNA_snn_res.0.4"] <- "res"int_cells.metadata <- int_cells@meta.dataint_cells.metadata <- int_cells.metadata[,-5]int_cells.metadata <- int_cells.metadata[,-4]int_cells.metadata$orig.ident <- "intermediate"names(int_cells.metadata)[names(int_cells.metadata) == "RNA_snn_res.0.4"] <- "res"progen.metadata <- progen@meta.dataprogen.metadata <- progen.metadata[,-5]progen.metadata <- progen.metadata[,-4]progen.metadata$orig.ident <- "progenitors"names(progen.metadata)[names(progen.metadata) == "RNA_snn_res.0.4"] <- "res"list_cells <- list(term_cells.matrix,int_cells.matrix,progen.matrix)combined_matrix <- dsCombineDGE(list_cells)combined_metadata <- do.call("rbind", list(progen.metadata,int_cells.metadata,term_cells.metadata))#create an URD object#which will filter the data, then normalize and log-transform it.cnido_URD <- createURD(count.data = combined_matrix, meta = combined_metadata, min.cells=5, min.counts=200)# Copy stage from @meta to @group.ids #this takes in the stage columns in the metadata, and placing it into a new slot in the data object called stage within group.ids. cnido_URD@group.ids$stage <- as.character(cnido_URD@meta[rownames(cnido_URD@group.ids),"orig.ident"])cnido_URD@group.ids#taking the stage names that you had in your group.ids slot in the URD object, and creating a new object called stagesstages <- sort(unique(cnido_URD@group.ids$stage))stages#you are taking the variable genes per stage
# diffCV.cutoff can be varied to include more or fewer genesvar.by.stage <- lapply(seq(1,3,1), function(n) {findVariableGenes(cnido_URD, cells.fit=cellsInCluster(cnido_URD, "stage", stages), set.object.var.genes=F, diffCV.cutoff=0.3, mean.min=.005, mean.max=100, main.use=paste0("Stages ", stages[n-2], " to ", stages[n]), do.plot=T)})# Combine results from each group of stages into a single list of variable genes and load into the URD objectvar.genes <- sort(unique(unlist(var.by.stage)))cnido_URD@var.genes <- var.genes# Calculate PCA and consider those PCs that with standard deviation 2x expected by noise as significantcnido_URD <- calcPCA(cnido_URD, mp.factor = 2)pcSDPlot(cnido_URD)# Calculate tSNEset.seed(19)cnido_URD <- calcTsne(object = cnido_URD)plotDim(cnido_URD, "orig.ident", plot.title = "tSNE: Stage")#sqrt number of cells (2463) = 50cnido_URD <- calcDM(cnido_URD, knn = 50, sigma=NULL)
#NULL sigma = 26.447plotDimArray(cnido_URD, reduction.use = "dm", dims.to.plot = 1:18, outer.title = "Diffusion Map(Sigma NULL 100NNs): Stage", label="stage", plot.title="", legend=F)plotDim(cnido_URD, "orig.ident", transitions.plot = 10000, plot.title="Developmental stage (with transitions)")#calculate pseudotime# Here we use all cells from the progenitors stage as the rootroot.cells <- cellsInCluster(cnido_URD, "stage", "progenitors")# Then we run 'flood' simulationscnido_URD.floods <- floodPseudotime(cnido_URD, root.cells = root.cells, n=50, minimum.cells.flooded = 2, verbose=F)# The we process the simulations into a pseudotimecnido_URD <- floodPseudotimeProcess(cnido_URD, cnido_URD.floods, floods.name="pseudotime")pseudotimePlotStabilityOverall(cnido_URD)plotDim(cnido_URD, "pseudotime")# Create a subsetted object of just those cells from the terminal stageterm.URD <- urdSubset(cnido_URD, cells.keep=cellsInCluster(cnido_URD, "stage", "terminal"))# Use the variable genes that were calculated only on the terminal stages term.URD@var.genes <- var.by.stage[[3]]# Calculate PCA and tSNEterm.URD <- calcPCA(term.URD, mp.factor = 1.5)pcSDPlot(term.URD)set.seed(20)term.URD <- calcTsne(term.URD, which.dims = 1:10, perplexity = 50)plotDim(term.URD, "res")#biased random walkscnido_URD@group.ids[rownames(term.URD@group.ids), "tip.clusters"] <- term.URD@meta$`res`#cell numbers may need to be modified for larger or smaller data setscnido_URD.ptlogistic <- pseudotimeDetermineLogistic(cnido_URD, "pseudotime", optimal.cells.forward=20, max.cells.back=40, do.plot = T)# Bias the transition matrix according to pseudotimecnido_URD.biased.tm <-as.matrix(pseudotimeWeightTransitionMatrix(cnido_URD, "pseudotime", logistic.params=cnido_URD.ptlogistic))# Simulate the biased random walks from each tipcnido_URD.walks <- simulateRandomWalksFromTips(cnido_URD, tip.group.id="tip.clusters", root.cells=root.cells, transition.matrix = cnido_URD.biased.tm, n.per.tip = 25000, root.visits = 1, max.steps = 5000, verbose = F)# Process the biased random walks into visitation frequenciescnido_URD <- processRandomWalksFromTips(cnido_URD, cnido_URD.walks, verbose = F)# Load the cells used for each tip into the URD objectcnido_URD.tree <- loadTipCells(cnido_URD, "tip.clusters")# Build the treecnido_URD.tree <- buildTree(cnido_URD.tree, pseudotime = "pseudotime", tips.use= c("3", "4", "5", "8"), divergence.method = "preference", cells.per.pseudotime.bin = 25, bins.per.pseudotime.window = 8, save.all.breakpoint.info = T, p.thresh=0.001)plotTree(cnido_URD.tree, "segment", label.segments = T)plotTree(cnido_URD.tree, "stage", label.segments = T)spiros_tip <- c("10")spiro_markers <- aucprTestAlongTree(cnido_URD.tree, pseudotime="pseudotime", tips=spiros_tip, log.effect.size=0.4, auc.factor = 1.25, max.auc.threshold = 0.85, frac.must.express = 0.1, frac.min.diff = 0, genes.use=genes.use, root="11", only.return.global=F, must.beat.sibs=0.1, report.debug=T, segs.to.skip = NULL)write.table(spiro_markers$diff.exp, file = "cnido_URD_spiro_DE.csv")nem_tip <- c("9")nem_markers <- aucprTestAlongTree(cnido_URD.tree, pseudotime="pseudotime", tips=nem_tip, log.effect.size=0.4, auc.factor = 1.25, max.auc.threshold = 0.85, frac.must.express = 0.1, frac.min.diff = 0, genes.use=genes.use, root="11", only.return.global=F, must.beat.sibs=0.1, report.debug=T, segs.to.skip = NULL)write.table(nem_markers$diff.exp, file = "cnido_URD_nem_DE.csv")cnido_URD.tree <- nameSegments(cnido_URD.tree, segments=c("11","9","10", "3", "5", "4", "8"), segment.names = c("progen","nem.progen","spiro.progen", "nem1", "nem2", "robust.spiro", "gracile.spiro"))plotTree(cnido_URD.tree, "segment", label.segments = F, color.tree = TRUE, discrete.colors = c("orchid2", "goldenrod1", "royalblue", "hotpink", "cyan3", "mediumpurple", "seagreen3"))
