Apoc_sc <- readRDS(file = "Apoc_sc_seurat.rds")

DimPlot(Apoc_sc, reduction = "umap", label = "TRUE")


#subset out the progenitor cells - cluster 1 that is made up of all 12hpf cells
progen <- subset(Apoc_sc, idents = c("blastula"))

#subet out terminal cells - not going to include mature cnido cluster 21 becaue that might confuse URD 


term_cells <- subset(Apoc_sc, idents = c("gastrodermis", "mucous.gland", "neural", "secretory.3", "gland.mouth", "secretory.1", "gland.1", "secretory.2", "spirocytes", "nem.1", "nem.2"))


#subset out intermediate cell clusters
int_cells <- subset(Apoc_sc, idents = c("endoderm.progen", "progen.1", "progen.2", "sec.progen.1", "sec.progen.2", "neurogland.progen", "cnido.progen"))

#extract cell population data as matricies
term_cells.matrix <- GetAssayData(object = term_cells, assay = "RNA", layer = "data")
int_cells.matrix <- GetAssayData(object = int_cells, assay = "RNA", layer = "data")
progen.matrix <- GetAssayData(object = progen, assay = "RNA", layer = "data")

#extract cell population specific metadata
term_cells.metadata <- term_cells@meta.data
term_cells.metadata$orig.ident <- "terminal"
names(term_cells.metadata)[names(term_cells.metadata) == "RNA_snn_res.0.5"] <- "res"
term_cells.metadata <- term_cells.metadata[,-4]

int_cells.metadata <- int_cells@meta.data
int_cells.metadata$orig.ident <- "intermediate"
names(int_cells.metadata)[names(int_cells.metadata) == "RNA_snn_res.0.5"] <- "res"
int_cells.metadata <- int_cells.metadata[,-4]

progen.metadata <- progen@meta.data
progen.metadata$orig.ident <- "progenitors"
names(progen.metadata)[names(progen.metadata) == "RNA_snn_res.0.5"] <- "res"
progen.metadata <- progen.metadata[,-4]

list_cells <- list(term_cells.matrix,int_cells.matrix,progen.matrix)
combined_matrix <- dsCombineDGE(list_cells)

combined_metadata <- do.call("rbind", list(progen.metadata,int_cells.metadata,term_cells.metadata))

#create an URD object
#which will filter the data, then normalize and log-transform it.
Apoc_URD <- createURD(count.data = combined_matrix, meta = combined_metadata, min.cells=5, min.counts=200)

# Copy stage from @meta to @group.ids 
#this takes in the orig.idents columns in the metadata, and placing it into a new slot in the axial data object called stage within group.ids. 
Apoc_URD@group.ids$stage <- as.character(Apoc_URD@meta[rownames(Apoc_URD@group.ids),"orig.ident"])
Apoc_URD@group.ids

# Get variable genes for each group of 3 stages
#now you are taking the stage names that you had in your group.ids slot in the URD object, and creating a new object called stages. This takes all of the unique stages and then creates
stages <- sort(unique(Apoc_URD@group.ids$stage))
stages

#you are taking the variable genes per stage here. 
var.by.stage <- lapply(1:length(stages), function(n) findVariableGenes(Apoc_URD, cells.fit=cellsInCluster(Apoc_URD, "stage", stages[[n]]), set.object.var.genes=F, diffCV.cutoff=0.3, mean.min=0.005, mean.max=100, main.use = stages[[n]], do.plot=T))

# Combine the results from each group of stages into a single list of variable genes and load into the URD object
var.genes <- sort(unique(unlist(var.by.stage)))
Apoc_URD@var.genes <- var.genes

saveRDS(Apoc_URD, "Apoc_URD_prePCA_var_4-28-26")

#Apoc_URD_var_walks.R script:
library(URD)

Apoc_URD_var <- readRDS(file = "Apoc_URD_var_walks.rds")

Apoc_URD_var.ptlogistic <- pseudotimeDetermineLogistic(Apoc_URD_var, "pseudotime", optimal.cells.forward=40, max.cells.back=80, do.plot = T)

# Bias the transition matrix acording to pseudotime
Apoc_URD_var.biased.tm <-as.matrix(pseudotimeWeightTransitionMatrix(Apoc_URD_var, "pseudotime", logistic.params=Apoc_URD_var.ptlogistic))

root.cells <- rownames(Apoc_URD_var@group.ids)[Apoc_URD_var@group.ids$stage == "progenitors"]

# Simulate the biased random walks from each tip
Apoc_URD_var.walks <- simulateRandomWalksFromTips(Apoc_URD_var, tip.group.id="tip.clusters", root.cells=root.cells, transition.matrix = Apoc_URD_var.biased.tm, n.per.tip = 25000, root.visits = 1, max.steps = 5000, verbose = F)

# Process the biased random walks into visitation frequencies
Apoc_URD_var <- processRandomWalksFromTips(Apoc_URD_var, Apoc_URD_var.walks, verbose = F)

# Load the cells used for each tip into the URD object
Apoc_URD_var.tree <- loadTipCells(Apoc_URD_var, "tip.clusters")

# Build the tree
Apoc_URD_var.tree <- buildTree(Apoc_URD_var.tree, pseudotime = "pseudotime", tips.use=c("8", "9", "7", "23", "12", "13", "16", "17", "18", "15", "19"), divergence.method = "preference", cells.per.pseudotime.bin = 80, bins.per.pseudotime.window = 5, save.all.breakpoint.info = T, p.thresh=0.001)

saveRDS(Apoc_URD_var, file = "Apoc_URD_var_processed_walks.rds")
saveRDS(Apoc_URD_var.tree, file = "Apoc_URD_var_tree.rds")
